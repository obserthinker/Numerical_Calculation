\section{等价关系与集合的分类}
\subsection{等价关系}
\begin{definition}\label{关系}
  关系\quad 设S是一个非空集合，\textrm{R}是关于S的元素的一个条件，如果对S中任意一个有序元素对(a,b)，都能确定a and b 是否满足条件\textrm{R}，那么称\textrm{R}是S上的一个关系。如果a,b满足条件就称为a,b有关系R,记作aRb，否则称为无关系R。关系R也叫做二元关系。
\end{definition}

\begin{definition}\label{等价关系}
  等价关系\quad R为S的一个关系，如果R满足三个条件\par
  \begin{enumerate}
    \item 反身性
    \item 对称性
    \item 传递性
  \end{enumerate}
  R为S的一个等价关系。
\end{definition}

\begin{definition}\label{等价类和商集}
  等价类和商集\quad 设$\sim$是S的一个等价关系，那么$a\in S$,令
  \begin{equation*}
    [a]={x\in S|x~a}
  \end{equation*}
  则[a]为S的一个等价类。S的全体等价类的集合称为集合S在等价关系的商集，记作S/$\sim$.
\end{definition}
